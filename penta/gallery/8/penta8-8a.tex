\newcommand{\clusterI}[8]{ % p,x,y,rot,mirror,P1,P2,P3
 \def\p{#1}
 \pgfmathsetmacro\rotI{-atan((5/6)/(7*sqrt(6^2 - 2^2)/6)}
 \begin{scope}[shift={(#2,#3)},rotate=(\rotI*#5)+#4] % displace,rotate
  \begin{scope}[yscale=#5] % mirror
   \begin{scope}[rotate=-asin(1/3)]
    \meccanostrip[0000FF]{7}{1}{\p}
    \path(4,0) ++(-90:20+\p) node{$#6$}; %H
    \path(7,0) ++(0:20+\p) node{$#7$}; %I
    \begin{scope}[shift={(4,0)},rotate=2*asin(1/3)]
     \meccanostrip[0088FF]{3}{1}{\p}
    \end{scope}
    \begin{scope}[shift={(7,0)},rotate=90+asin(1/3)]
     \meccanostrip[0088FF]{3}{1}{\p}
     \path(2,0) ++(-130:4+20+\p) node{$#8$}; %J
    \end{scope}
   \end{scope}
  \end{scope}
 \end{scope}
}


\newcommand{\clusterII}[7]{ % p,x,y,rot,mirror,P1,P2
 \def\p{#1}
 \pgfmathsetmacro\rotII{-atan(sqrt(2^2 - (3/2)^2)/(13/2)}
 \begin{scope}[shift={(#2,#3)},rotate=(\rotII*#5)+#4] % displace,rotate
  \begin{scope}[yscale=#5] % mirror
   \meccanostrip[FF4400]{8}{1}{\p}
   \path (8,0) ++(0:20+\p) node{$#6$};
   \begin{scope}[shift={(5,0)},rotate=asin(2/3)]
    \meccanostrip[880088]{2}{1}{\p}
    \path (4,0) ++(40:20+\p) node{$#7$};
   \end{scope}
   \begin{scope}[shift={(8,0)},rotate=180-asin(2/3)]
    \meccanostrip[880088]{2}{1}{\p}
   \end{scope}
  \end{scope}
 \end{scope}
}
