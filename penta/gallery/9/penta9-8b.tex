\newcommand{\clusterII}[7]{ % p,x,y,rot,mirror,P1,P2
 \def\p{#1}
 \pgfmathsetmacro\rot{acos((61+64-1)/(2*sqrt(61)*8))}
 \begin{scope}[shift={(#2,#3)},rotate=(\rot*#5)+#4] % displace,rotate
  \begin{scope}[yscale=#5] % mirror
   \meccanostrip[0000FF]{8}{1}{\p}
   \path (8,0) ++(40:20+\p) node{$#6$};
   \begin{scope}[shift={(8,0)},rotate=180+acos(1/4)]
    \meccanostrip[0088FF]{1}{1}{\p}
    \begin{scope}[shift={(1,0)},rotate=180+acos(1/4)]
     \meccanostrip[0000FF]{2}{1}{\p}
     \path (2,0) ++(90:20+\p) node{$#7$};
    \end{scope}
   \end{scope}
  \end{scope}
 \end{scope}
}


\newcommand{\clusterIII}[7]{ % p,x,y,rot,mirror,P1,P2

 \begin{scope}[shift={(#2,#3)},rotate=(\rot*#5)+#4] % displace,rotate
  \begin{scope}[yscale=#5] % mirror
   \meccanostrip[0000FF]{8}{1}{\p}
   \path (8,0) ++(0:20+\p) node{$#6$};
   \begin{scope}[shift={(8,0)},rotate=180+acos(1/4)]
    \meccanostrip[0088FF]{4}{1}{\p}
    \path (4,0) ++(40:20+\p) node{$#7$};
    \begin{scope}[shift={(4,0)},rotate=180+acos(1/4)]
     \meccanostrip[0000FF]{8}{1}{\p}
    \end{scope}
   \end{scope}
  \end{scope}
 \end{scope}
}
