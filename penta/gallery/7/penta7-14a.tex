\newcommand{\cluster}[9]{ %p,dx,rot
 \def\p{#1}
 \def\r{12+\p}
 \pgfmathsetmacro\cx{sqrt(4^2 - 1^2) + sqrt(7^2 - (7/2)^2)}
 \pgfmathsetmacro\cy{7/2}
 \pgfmathsetmacro\crot{0}%-atan(\cy/\cx)}
 \begin{scope}[shift={(#2,0)},rotate=\crot+#3]
  \begin{scope}[rotate=asin(2/3)]
   \path(3,0) ++(-135:3+\r) node{$#4$}; %H
   \path(6,0) ++(200:12+\r) node{$#5$}; %I
   \meccanostrip[0088FF]{6}{1}{\p}
   \begin{scope}[shift={(3,0)},rotate=-2*asin(2/3)]
    \meccanostrip[00CC00]{3}{1}{\p}
   \end{scope}   
   \begin{scope}[shift={(6,0)},rotate=180+acos(2/3)]
    \meccanostrip[880088]{4}{1}{\p}
    \path(4,0) ++(-45:\r) node{$#6$}; %J
    \path(2,0) ++(-90:\r) node{$#7$}; %K
    \begin{scope}[shift={(4,0)},rotate=90]
     \meccanostrip[0000FF]{2}{1}{\p}
     \begin{scope}[shift={(2,0)},rotate=180-45-acos(5/(4*sqrt(2)))]
      \meccanostrip[880088]{4}{1}{\p}
      \path(1,0) ++(-90:\r) node{$#8$}; %L
      \path(4,0) ++(-110:\r) node{$#9$}; %M
      \begin{scope}[shift={(1,0)},rotate=asin(2/3)]
       \meccanostrip[0000FF]{2}{1}{\p}
      \end{scope}
      \begin{scope}[shift={(4,0)},rotate=180-asin(2/3)]
       \meccanostrip[0000FF]{2}{1}{\p}
      \end{scope}
     \end{scope}
    \end{scope}
   \end{scope}
  \end{scope}
 \end{scope}
}
