\tikzmath{
	\a = #1; \b = #2; \c = #3; \f = #4; \sx= #5; \sy= #6;
	\aa = 1/\a;
	\bb = 1/\b;
	\cc = 1/\c;
	\d = int(\a - 2*\b);
	\dd = 1/\d);
	\A = 180/7;
	\cosA = cos(1*\A); \sinA = sin(1*\A);
	\cosB = cos(2*\A); \sinB = sin(2*\A);
	\cosC = cos(3*\A); \sinC = sin(3*\A);
	\r = 0.2/(\a*\f);
	coordinate \pA, \pB, \pC, \pD, \pE, \pF, \pG, \pH, \pI, \pJ, \pK, \pL, \pM;
	\pA = (0,0);
	\pB = (\a,0);
	\pC = (\pB) + ( \b*\cosB,  \b*\sinB);
	\pD = (\pC) + ( \d*\cosB,  \d*\sinB);
	\pE = (\pD) + ( \b*\cosB,  \b*\sinB);
	\pF = (\pE) + (-\a*\cosC,  \a*\sinC);
	\pG = (\pF) + (-\a*\cosA,  \a*\sinA);
	\pH = (\pG) + (-\a*\cosA, -\a*\sinA);
	\pI = (\pH) + (-\a*\cosC, -\a*\sinC);
	\pJ = (\pI) + ( \b*\cosB, -\b*\sinB);
	\pK = (\pJ) + ( \d*\cosB, -\d*\sinB);
	%
	\pL = (\pG) + ( \a*\cosC, -\a*\sinC);
	\pM = (\pG) + (-\a*\cosC, -\a*\sinC);
}
\begin{scope}[shift={(\sx,\sy)},scale={\f}]

-- (\pA) foreach \t in {0,\bb,...,1} { pic [pos=\t] {code={\fill circle[radius=\r];}}} node[midway,below left,black]{\b}

  \draw[blue] (\pE)

  \draw[green] (\pJ)

\end{scope}
